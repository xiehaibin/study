1. bad()        // 如果进行非法操作 返回true 否则返回 false

2. clear()      // 设置内部错误状 如果用缺省参量调用则清除所有错误位 只有一部分编绎器要qo在第二次打开                // 同一个文件进行读取之前调用clear();

3. eof()        // 如果提取操作已经到达文件尾 返回true, 否则返回 false

4. good()       // 如果没有错误条件和没有设置文件结束标志， 返回true， 否则返回 false

5.fail()        // 与good相反 操作失败返回false 否则返回 true
 
6. is_open()    // 判断对象是否成功地与文件关联，若是 返回true, 否则返回 false
